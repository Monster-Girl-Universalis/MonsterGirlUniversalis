unlock_race = {
	if = {
		limit = { NOT = { has_race_unlocked = { race = $race$ } } }
		custom_tooltip = unlock_race_$race$
		set_country_flag = can_access_race_$race$
	}
}

###CHAOS VARIABLES###
MGQ_add_chaos = {
	if = {
		limit = {
			ai = no
		}
		change_variable = {
			which = MGQ_chaos
			value = $value$
		}
		hidden_effect = {
			if = {
				limit = {
					check_variable = {
						which = MGQ_chaos
						value = 101
					}
				}
				set_variable = {
					which = MGQ_chaos
					value = 100
				}
			}
		}
	}
}

unlock_centaur_cavalry = {
		custom_tooltip = unlock_centaur_cavalry
		set_country_flag = can_access_centaur_cavalry
}

MGQ_reduce_chaos = {
	if = {
		limit = {
			ai = no
		}
		subtract_variable = {
			which = MGQ_chaos
			value = $value$
		}
		if = {
			limit = {
				NOT = {
					check_variable = {
						which = MGQ_chaos
						value = 0
					}
				}
			}
			set_variable = {
				which = MGQ_chaos
				value = 0
			}
		}
	}
}

MGQ_add_yearly_chaos = {
	hidden_effect = {
		export_to_variable = {
			which = yearly_chaos_variable
		}
		export_to_variable = {
			which = monthly_battle_passion_variable
			value = modifier:yearly_chaos
		}
	}
}
