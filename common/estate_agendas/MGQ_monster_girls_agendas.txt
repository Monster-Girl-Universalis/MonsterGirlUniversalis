#Give us 3 privileges
estate_monster_girls_three_privileges = {
	can_select = {
		NOT = {
			num_of_estate_privileges = {
				estate = estate_monster_girls
				value = 2
			}
		}
	}
	selection_weight = {
		factor = 1
	}
	pre_effect = {
		set_variable = {
			which = estate_monster_girls_three_privileges_var
			value = 0
		}
		if = {
			limit = {
				num_of_estate_privileges = {
					estate = estate_monster_girls
					value = 1
				}
			}
			set_variable = {
				which = estate_monster_girls_three_privileges_var
				value = 2
			}
		}
		else = {
			set_variable = {
				which = estate_monster_girls_three_privileges_var
				value = 1
			}
		}
	}
	task_requirements = {
		if = {
			limit = {
				check_variable = {
					which = estate_monster_girls_three_privileges_var
					value = 2
				}
			}
			num_of_estate_privileges = {
				estate = estate_monster_girls
				value = 3
			}
		}
		else = {
			num_of_estate_privileges = {
				estate = estate_monster_girls
				value = 2
			}
		}
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_monster_girls
			loyalty = 15
		}
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_monster_girls
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -10
			duration = 7300
        }
    }
}

#Mermaids want access to the Sea
estate_monster_girls_mermaids_access_sea = {
	can_select = {
		is_free_or_tributary_trigger = yes
		NOT = {
			any_owned_province = {
				has_port = yes
			}
		}
		any_owned_province = {
			any_neighbor_province = {
				has_port = yes
				owner = {
					NOT = { overlord_of = root }
					NOT = { is_subject_of = root }
					NOT = { alliance_with = root }
					NOT = { ROOT = { truce_with = prev } }
					NOT = {
						has_opinion = {
							who = root
							value = 50
						}
					}
					OR = {
						is_part_of_hre = no
						NOT = { hre_reform_passed = landfriede }
						ROOT = { is_part_of_hre = no }
					}
				}
			}
		}
	}
	selection_weight = {
		factor = 2.5
		modifier = {
			factor = 0.25
			any_country = {
				coalition_target = root
			}
		}
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				any_neighbor_province = {
					has_port = yes
					owner = {
						NOT = { overlord_of = root }
						NOT = { is_subject_of = root }
						NOT = { alliance_with = root }
						NOT = { ROOT = { truce_with = prev } }
						NOT = {
							has_opinion = {
								who = root
								value = 50
							}
						}
						OR = {
							is_part_of_hre = no
							NOT = { hre_reform_passed = landfriede }
							ROOT = { is_part_of_hre = no }
						}
					}
				}
			}
			random_neighbor_province = {
				limit = {
					has_port = yes
					owner = {
						NOT = { overlord_of = root }
						NOT = { is_subject_of = root }
						NOT = { alliance_with = root }
						NOT = { ROOT = { truce_with = prev } }
						NOT = {
							has_opinion = {
								who = root
								value = 50
							}
						}
						OR = {
							is_part_of_hre = no
							NOT = { hre_reform_passed = landfriede }
							ROOT = { is_part_of_hre = no }
						}
					}
				}
				save_event_target_as = agenda_province
			}
		}
	}
	provinces_to_highlight = {
		OR = {
			province_id = event_target:agenda_province
			AND = {
				OR = {
					is_claim = root
					is_core = root
				}
				has_port = yes
			}
		}
	}
	task_requirements = {
		any_owned_province = {
			has_port = yes
		}
	}
	immediate_effect = {
		if = {
			limit = {
				event_target:agenda_province = {
					NOT = { is_core = root }
					NOT = { is_claim = root }
				}
			}
			event_target:agenda_province = {
				add_claim = root
			}
		}
	}
	on_invalid = {
		event_target:agenda_province = {
			if = {
				limit = {
					is_claim = root
					NOT = { is_permanent_claim = root }
				}
				remove_claim = root
			}
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_monster_girls
			loyalty = 15
		}
		add_sailors = 500
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_monster_girls
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -10
			duration = 7300
        }
		event_target:agenda_province = {
			if = {
				limit = {
					is_claim = root
					NOT = { is_permanent_claim = root }
				}
				remove_claim = root
			}
		}
    }
}
