#Give us 3 privileges
estate_angels_three_privileges = {
	can_select = {
		NOT = {
			num_of_estate_privileges = {
				estate = estate_angels
				value = 2
			}
		}
	}
	selection_weight = {
		factor = 1
	}
	pre_effect = {
		set_variable = {
			which = estate_angels_three_privileges_var
			value = 0
		}
		if = {
			limit = {
				num_of_estate_privileges = {
					estate = estate_angels
					value = 1
				}
			}
			set_variable = {
				which = estate_angels_three_privileges_var
				value = 2
			}
		}
		else = {
			set_variable = {
				which = estate_angels_three_privileges_var
				value = 1
			}
		}
	}
	task_requirements = {
		if = {
			limit = {
				check_variable = {
					which = estate_angels_three_privileges_var
					value = 2
				}
			}
			num_of_estate_privileges = {
				estate = estate_angels
				value = 3
			}
		}
		else = {
			num_of_estate_privileges = {
				estate = estate_angels
				value = 2
			}
		}
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_angels
			loyalty = 15
		}
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_angels
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -10
			duration = 7300
        }
    }
}
estate_angels_develop_x = {
	selection_weight = {
		factor = 1
	}
	can_select = {
		any_owned_province = {
			is_state_core = root
		}
	}
	pre_effect = {
		set_variable = {
			which = develop_x_category_angels
			value = 0
		}
		random_list = {
			1 = { 
				set_variable = {
					which = develop_x_category_angels
					value = 3
				}
			}
			2 = { #dip
				set_variable = {
					which = develop_x_category_angels
					value = 2
				}
			}
			1 = { #adm
				set_variable = {
					which = develop_x_category_angels
					value = 1
				}
			}
		}
		random_owned_province = {
			limit = {
				is_state_core = root
			}
			save_event_target_as = agenda_province
		}
		event_target:agenda_province = {
			if = {
				limit = {
					root = {
						check_variable = {
							which = develop_x_category_angels
							value = 3
						}
					}
				}
				if = {
					limit = {
						check_variable = {
							which = province_mil_var
							value = 1
						}
					}
					set_variable = {
						which = province_mil_var
						value = 0
					}
				}
				export_to_variable = {
					which = province_mil_var
					value = trigger_value:base_manpower
				}
				change_variable = {
					which = province_mil_var
					value = 2
				}
			}
			else_if = {
				limit = {
					root = {
						check_variable = {
							which = develop_x_category_angels
							value = 2
						}
					}
				}
				if = {
					limit = {
						check_variable = {
							which = province_dip_var
							value = 1
						}
					}
					set_variable = {
						which = province_dip_var
						value = 0
					}
				}
				export_to_variable = {
					which = province_dip_var
					value = trigger_value:base_production
				}
				change_variable = {
					which = province_dip_var
					value = 2
				}
			}
			else = {
				if = {
					limit = {
						check_variable = {
							which = province_adm_var
							value = 1
						}
					}
					set_variable = {
						which = province_adm_var
						value = 0
					}
				}
				export_to_variable = {
					which = province_adm_var
					value = trigger_value:base_tax
				}
				change_variable = {
					which = province_adm_var
					value = 2
				}
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}
    task_requirements = {
		event_target:agenda_province = {
			if = {
				limit = {
					root = {
						check_variable = {
							which = develop_x_category_angels
							value = 3
						}
					}
				}
				base_manpower = "province_mil_var"
			}
			else_if = {
				limit = {
					root = {
						check_variable = {
							which = develop_x_category_angels
							value = 2
						}
					}
				}
				base_production = "province_dip_var"
			}
			else = {
				base_tax = "province_adm_var"
			}
		}
    }
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_angels
			loyalty = 15
		}
		if = {
			limit = {
				check_variable = {
					which = develop_x_category_angels
					value = 3
				}
			}
			 add_yearly_manpower = 0.3
		}
		else_if = {
			limit = {
				check_variable = {
					which = develop_x_category_angels
					value = 2
				}
			}
			add_years_of_income = 0.3
		}
		else = {
			add_prestige = 10
		}
    }
    failing_effect = {
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_named_unrest = {
					name = local_angels_displeased
					value = 5
				}
			}
		}
        add_estate_loyalty_modifier = {
            estate = estate_angels
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -10
			duration = 7300
        }
    }
}

#Convert province culture
estate_angels_convert_province_culture = {
	can_select = {
		any_owned_province = {
			NOT = { culture = root }
			has_owner_accepted_culture = no
			NOT = { nationalism = 1 }
			religion = root
			any_neighbor_province = {
				owned_by = root
				culture = root
			}
		}
		NOT = {
			tag = ENV
			was_tag = ENV
			culture = Fallen_Angel
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.75
			always = yes
		}
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				NOT = { culture = root }
				has_owner_accepted_culture = no
				NOT = { nationalism = 1 }
				religion = root
				any_neighbor_province = {
					owned_by = root
					culture = root
				}
			}
			save_event_target_as = angels_agenda_province
			choose_random_development_bonus_effect = yes
		}
	}
	fail_if = {
		event_target:angels_agenda_province = {
			OR = {
				NOT = { owned_by = root }
				nationalism = 1
				NOT = { religion = root }
			}
		}
	}
	immediate_effect = {
		event_target:angels_agenda_province = {
			add_province_modifier = {
				name = agenda_cultural_conversion_help
				duration = 7300
			}
		}
	}
	on_invalid = {
		event_target:angels_agenda_province = {
			clear_random_development_bonus_flags_effect = yes
			if = {
				limit = {
					has_province_modifier = agenda_cultural_conversion_help
				}
				remove_province_modifier = agenda_cultural_conversion_help
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:angels_agenda_province
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_angels
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_angels
			}
		}
		else = {
			if = {
				limit = {
					NOT = {
						event_target:angels_agenda_province = {
							has_owner_accepted_culture = yes
						}
					}
				}
				event_target:angels_agenda_province = {
					culture = root
				}
			}
			else = { #don't want to show this in most cases, but don't want it to count as a fail if you do this.
				event_target:angels_agenda_province = {
					has_owner_accepted_culture = yes
				}
			}
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_angels
		}
		add_estate_loyalty = {
			estate = estate_angels
			loyalty = 15
		}
		event_target:angels_agenda_province = {
			apply_random_development_bonus_bonus_effect = yes
		}
		hidden_effect = {
			event_target:angels_agenda_province = {
				if = {
					limit = {
						has_province_modifier = agenda_cultural_conversion_help
					}
					remove_province_modifier = agenda_cultural_conversion_help
				}
			}
		}
	}
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_angels
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -10
			duration = 7300
        }
		event_target:angels_agenda_province = {
			clear_random_development_bonus_flags_effect = yes
			if = {
				limit = {
					has_province_modifier = agenda_cultural_conversion_help
				}
				remove_province_modifier = agenda_cultural_conversion_help
			}
		}
    }
}


#Increase autonomy in X
estate_angels_increase_autonomy_in_province_x = {
	can_select = {
		any_owned_province = {
			is_state_core = root
			NOT = { local_autonomy = 75 }
			OR = {
				tag = ENV
				was_tag = ENV
				culture = Fallen_Angel
			}
			is_capital = no
			NOT = { has_province_modifier = autonomy_increased }
			NOT = { has_province_modifier = autonomy_decreased }
		}
	}
	selection_weight = {
		factor = 1.5
	}
	pre_effect = {
		set_variable = {
			which = estate_angels_increase_autonomy_in_province_x_var
			value = 0
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { local_autonomy = 75 }
				is_capital = no
				NOT = { has_province_modifier = autonomy_increased }
				NOT = { has_province_modifier = autonomy_decreased }
			}
			save_event_target_as = agenda_province
			choose_random_development_bonus_effect = yes
		}
		event_target:agenda_province = {
			if = {
				limit = {
					local_autonomy = 60
				}
				root = {
					set_variable = {
						which = estate_angels_increase_autonomy_in_province_x_var
						value = 5
					}
				}
			}
			else_if = {
				limit = {
					local_autonomy = 45
				}
				root = {
					set_variable = {
						which = estate_angels_increase_autonomy_in_province_x_var
						value = 4
					}
				}
			}
			else_if = {
				limit = {
					local_autonomy = 30
				}
				root = {
					set_variable = {
						which = estate_angels_increase_autonomy_in_province_x_var
						value = 3
					}
				}
			}
			else_if = {
				limit = {
					local_autonomy = 15
